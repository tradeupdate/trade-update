async function fetchBacktestCandles(
  symbol: string,
  dateFrom: number,
  dateTo: number
): Promise<Candle[]> {

  const allCandles: Candle[] = []
  const CHUNK_SIZE = 5000
  const GRANULARITY = 300 // 5m candles
  
  let currentEnd = dateTo
  let chunksAttempted = 0
  const MAX_CHUNKS = 10

  console.log(`Fetching ${symbol} candles:
    From: ${new Date(dateFrom*1000).toISOString()}
    To: ${new Date(dateTo*1000).toISOString()}
    Expected: ~${Math.ceil(
      (dateTo-dateFrom)/(GRANULARITY)
    )} candles`)

  while (
    currentEnd > dateFrom && 
    chunksAttempted < MAX_CHUNKS
  ) {
    chunksAttempted++
    
    console.log(`Fetching chunk ${chunksAttempted}:
      End: ${new Date(currentEnd*1000).toISOString()}
      Candles so far: ${allCandles.length}`)

    const chunk = await fetchCandleChunk(
      symbol,
      GRANULARITY,
      CHUNK_SIZE,
      currentEnd
    )

    if (!chunk || chunk.length === 0) {
      console.log(`No more candles available 
        after chunk ${chunksAttempted}`)
      break
    }

    allCandles.push(...chunk)

    // Next chunk ends where this one started
    const earliestInChunk = Math.min(
      ...chunk.map(c => c.time)
    )
    currentEnd = earliestInChunk - GRANULARITY

    await sleep(300) // Respect rate limits

    if (earliestInChunk <= dateFrom) {
      console.log(`Reached target start date`)
      break
    }
  }

  // Filter to exact date range
  const filtered = allCandles.filter(
    c => c.time >= dateFrom && c.time <= dateTo
  )

  // Sort and deduplicate
  const sorted = filtered
    .sort((a,b) => a.time - b.time)
  
  const deduped = sorted.filter((c, i) =>
    i === 0 || c.time !== sorted[i-1].time
  )

  console.log(`Backtest candles ready:
    Total fetched: ${allCandles.length}
    After filter: ${filtered.length}
    After dedup: ${deduped.length}
    Date range: ${new Date(
      deduped[0]?.time*1000
    ).toISOString()} to ${new Date(
      deduped[deduped.length-1]?.time*1000
    ).toISOString()}`)

  return deduped
}

async function fetchCandleChunk(
  symbol: string,
  granularity: number,
  count: number,
  endTime: number
): Promise<Candle[]> {

  return new Promise((resolve, reject) => {
    const ws = new WebSocket(
      'wss://ws.binaryws.com/websockets/v3?app_id=1089'
    )

    const timeout = setTimeout(() => {
      ws.close()
      resolve([])
    }, 15000) // 15 second timeout per chunk

    ws.on('open', () => {
      ws.send(JSON.stringify({
        ticks_history: symbol,
        style: "candles",
        granularity,
        count,
        end: endTime
      }))
    })

    ws.on('message', (data: string) => {
      clearTimeout(timeout)
      
      const response = JSON.parse(data)
      
      if (response.error) {
        console.error(`Deriv error:`, response.error)
        ws.close()
        resolve([])
        return
      }

      if (response.candles) {
        const candles = response.candles.map(
          (c: any) => ({
            time: Number(c.epoch),
            open: Number(c.open),
            high: Number(c.high),
            low: Number(c.low),
            close: Number(c.close)
          })
        )
        ws.close()
        resolve(candles)
      }
    })

    ws.on('error', (err) => {
      clearTimeout(timeout)
      console.error(`WebSocket error:`, err)
      ws.close()
      resolve([])
    })
  })
}

interface CandleStore {
  candles1m: Candle[]
  candles5m: Candle[]
  candles15m: Candle[]
  candles1h: Candle[]
  candles4h: Candle[]
  lastUpdated: number
}

const candleStores: Map<string, CandleStore> 
  = new Map()

function initCandleStore(symbol: string): void {
  candleStores.set(symbol, {
    candles1m: [],
    candles5m: [],
    candles15m: [],
    candles1h: [],
    candles4h: [],
    lastUpdated: 0
  })
}

function addCandle(
  symbol: string,
  timeframe: string,
  candle: Candle
): void {
  const store = candleStores.get(symbol)
  if (!store) return

  const key = `candles${timeframe}` as keyof CandleStore
  const arr = store[key] as Candle[]

  // Find if this candle time already exists
  const existingIdx = arr.findIndex(
    c => c.time === candle.time
  )

  if (existingIdx >= 0) {
    // Update existing candle (still forming)
    arr[existingIdx] = candle
  } else {
    // New candle
    arr.push(candle)
    // Keep rolling window
    const maxCandles = {
      '1m': 1000,
      '5m': 1000,
      '15m': 500,
      '1h': 200,
      '4h': 100
    }
    const max = maxCandles[
      timeframe as keyof typeof maxCandles
    ] || 200
    
    if (arr.length > max) {
      arr.shift() // Remove oldest
    }
  }

  store.lastUpdated = Date.now()
}

function granularityToTf(g: number): string {
  const map: {[key:number]: string} = {
    60: '1m', 300: '5m', 900: '15m',
    3600: '1h', 14400: '4h'
  }
  return map[g] || '5m'
}

async function runBacktest(params: {
  userId: string
  strategyId: string
  dateFrom: string
  dateTo: string
}): Promise<BacktestResult> {

  const strategy = getStrategy(params.strategyId)
  const symbol = strategy.pair || 'R_75'
  
  // Convert dates to unix
  const fromUnix = normalizeDateToMidnightUTC(
    params.dateFrom
  )
  const toUnix = normalizeDateToMidnightUTC(
    params.dateTo
  ) + 86399 // End of day

  // Check cache first
  const cacheKey = 
    `${symbol}_5m_${fromUnix}_${toUnix}`
  const cached = loadFromCache(cacheKey)
  
  let candles5m: Candle[]
  
  if (cached) {
    console.log(`Using cached data: ${cacheKey}`)
    candles5m = cached.candles
  } else {
    console.log(`Fetching fresh data: ${cacheKey}`)
    candles5m = await fetchBacktestCandles(
      symbol, fromUnix, toUnix
    )
    saveToCache(cacheKey, candles5m)
  }

  console.log(`Backtest data ready:
    Symbol: ${symbol}
    5m candles: ${candles5m.length}
    Expected for date range: ~${Math.ceil(
      (toUnix - fromUnix) / 300
    )}`)

  // Build all derived timeframes
  const candles1h = build1hFromFiveMin(candles5m)
  const candles4h = build4hFromOneHour(candles1h)
  const candles15m = build15mFromFiveMin(candles5m)

  console.log(`Derived timeframes:
    15m: ${candles15m.length}
    1h: ${candles1h.length}
    4h: ${candles4h.length}`)

  // Route to correct backtest engine
  if (strategy.name.includes('Swing')) {
    return runSwingBacktest(
      candles5m, candles15m, candles1h, 
      candles4h, strategy, params
    )
  } else if (strategy.name.includes('Reversal')) {
    return runReversalBacktest(
      candles5m, candles15m, strategy, params
    )
  } else if (strategy.name.includes('V10')) {
    return runV10Backtest(
      candles5m, candles15m, strategy, params
    )
  } else {
    return runSniperBacktest(
      candles5m, candles15m, strategy, params
    )
  }
}
