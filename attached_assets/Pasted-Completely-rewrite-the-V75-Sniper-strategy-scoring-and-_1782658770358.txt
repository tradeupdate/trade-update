The old strategy tried to score everything 
and trade when score was high enough.
The problem: high scores don't predict 
direction. A perfect volatility score and 
perfect session score mean nothing if the 
trend is about to reverse.

New philosophy: THREE TIMEFRAME ALIGNMENT.

A trade only fires when ALL THREE timeframes
agree on the same direction simultaneously.

1h chart: defines the dominant trend
  (this is new - we weren't using 1h before)
15m chart: confirms trend is active
5m chart: provides the entry timing

If 1h says BULL, 15m says BULL, 5m says 
pullback complete = BUY
If any one of three disagrees = NO TRADE

This single rule eliminates the majority of
bad trades. A trade that looks good on 5m
but goes against the 1h trend is the most
common cause of losses on V75.

function build1hCandles(candles5m: Candle[]): Candle[] {
  const grouped: { [key: number]: Candle[] } = {}
  
  candles5m.forEach(candle => {
    // Round down to nearest hour boundary
    const boundary = Math.floor(candle.time / 3600) * 3600
    if (!grouped[boundary]) grouped[boundary] = []
    grouped[boundary].push(candle)
  })
  
  return Object.entries(grouped)
    .filter(([_, g]) => g.length > 0)
    .map(([boundary, group]) => {
      const sorted = group.sort((a,b) => a.time - b.time)
      return {
        time: parseInt(boundary),
        open: sorted[0].open,
        high: Math.max(...sorted.map(c => c.high)),
        low: Math.min(...sorted.map(c => c.low)),
        close: sorted[sorted.length-1].close
      }
    })
    .sort((a,b) => a.time - b.time)
}

COMPONENT 1 — 1H TREND (0-10 points)
This is the most important component.
If this is weak, nothing else matters.

Calculate on 1h candles:

- EMA50 on 1h (50 period exponential MA)
- Price position relative to both EMAs

function score1hTrend(candles1h, direction):
  closes = candles1h.map(c => c.close)
  ema20 = calculateEMA(closes, 20)
  ema50 = calculateEMA(closes, 50)
  currentPrice = closes[closes.length-1]
  
  if direction === "BUY":
    // All three must be aligned bullish
    if ema20 > ema50 AND currentPrice > ema20:
      // Check slope - EMA20 must be rising
      ema20_5bars_ago = calculateEMA(
        closes.slice(0, -5), 20)
      if ema20 > ema20_5bars_ago:
        return 10 // Perfect bull trend
      else:
        return 6 // Bull alignment but flat
    if ema20 > ema50 AND currentPrice < ema20:
      return 4 // Bull trend but price below EMA
    if ema20 < ema50:
      return 0 // Bear trend - NO BUY
  
  if direction === "SELL":
    if ema20 < ema50 AND currentPrice < ema20:
      ema20_5bars_ago = calculateEMA(
        closes.slice(0, -5), 20)
      if ema20 < ema20_5bars_ago:
        return 10 // Perfect bear trend
      else:
        return 6 // Bear alignment but flat
    if ema20 < ema50 AND currentPrice > ema20:
      return 4 // Bear trend but price above EMA
    if ema20 > ema50:
      return 0 // Bull trend - NO SELL

HARD RULE: If Component 1 scores 0 in the

function score15mConfirmation(candles15m, direction):
  closes15m = candles15m.map(c => c.close)
  ema9 = calculateEMA(closes15m, 9)
  ema21 = calculateEMA(closes15m, 21)
  adx = calculateADX(candles15m, 14)
  
  if direction === "BUY":
    if ema9 > ema21 AND adx > 25:
      return 10 // Strong bull confirmation
    if ema9 > ema21 AND adx > 20:
      return 7  // Moderate bull confirmation
    if ema9 > ema21 AND adx <= 20:
      return 3  // Weak - EMA aligned but no trend
    if ema9 < ema21:
      return 0  // 15m disagrees with 1h
  
  if direction === "SELL":
    if ema9 < ema21 AND adx > 25:
      return 10
    if ema9 < ema21 AND adx > 20:
      return 7
    if ema9 < ema21 AND adx <= 20:
      return 3
    if ema9 > ema21:
      return 0

HARD RULE: If Component 2 scores 0,
return NONE. Both 1h and 15m must agree.

COMPONENT 3 — 5M ENTRY TIMING (0-10 points)
Only reached if Components 1 and 2 pass.
Finds the precise moment to enter.

The best entry is a PULLBACK in a confirmed trend.
Not a breakout. Not chasing. A pullback.

Calculate on 5m candles:
- EMA21 on 5m (the pullback target)
- RSI(14) on 5m (momentum confirmation)
- Distance from current price to EMA21

function score5mEntry(candles5m, direction):
  closes5m = candles5m.map(c => c.close)
  ema21_5m = calculateEMA(closes5m, 21)
  rsi = calculateRSI(candles5m, 14)
  currentPrice = closes5m[closes5m.length-1]
  
  distanceFromEMA = Math.abs(currentPrice - ema21_5m)
  distancePct = distanceFromEMA / ema21_5m * 100
  
  if direction === "BUY":
    // We want price near or just touched EMA21
    // and RSI showing oversold on pullback
    
    priceNearEMA = distancePct < 0.15
    // Price within 0.15% of EMA21
    
    priceJustBelowEMA = currentPrice < ema21_5m 
      AND distancePct < 0.3
    // Just pulled back below EMA21
    
    rsiPullback = rsi > 30 AND rsi < 55
    // RSI in pullback territory, not oversold
    
    rsiOversold = rsi < 35
    // Stronger oversold = better entry
    
    if priceNearEMA AND rsiPullback:
      return 10 // Perfect pullback to EMA
    if priceNearEMA AND rsi < 60:
      return 8
    if priceJustBelowEMA AND rsiOversold:
      return 9 // Pulled back below EMA, oversold
    if distancePct < 0.5 AND rsi < 50:
      return 6 // Near EMA but not perfect
    if distancePct < 1.0 AND rsi < 45:
      return 4 // Somewhat near EMA
    if rsi < 30:
      return 5 // Oversold even if far from EMA
    
    // If price is far above EMA21 = chasing
    if currentPrice > ema21_5m AND distancePct > 0.5:
      return 0 // Do not chase a run
    
    return 2 // Default low score
  
  if direction === "SELL":
    // Mirror for sell direction
    priceNearEMA = distancePct < 0.15
    priceJustAboveEMA = currentPrice > ema21_5m 
      AND distancePct < 0.3
    rsiPullback = rsi > 45 AND rsi < 70
    rsiOverbought = rsi > 65
    
    if priceNearEMA AND rsiPullback:
      return 10
    if priceNearEMA AND rsi > 40:
      return 8
    if priceJustAboveEMA AND rsiOverbought:
      return 9
    if distancePct < 0.5 AND rsi > 50:
      return 6
    if distancePct < 1.0 AND rsi > 55:
      return 4
    if rsi > 70:
      return 5
    if currentPrice < ema21_5m AND distancePct > 0.5:
      return 0 // Do not chase a drop
    
    return 2

THRESHOLD = 20/30

function generateSignal(
  candles1h: Candle[],
  candles15m: Candle[],
  candles5m: Candle[]
): SignalResult {

  // Need minimum candle counts
  if (candles1h.length < 10) 
    return { signal: 'NONE', reason: 'Insufficient 1h data' }
  if (candles15m.length < 21) 
    return { signal: 'NONE', reason: 'Insufficient 15m data' }
  if (candles5m.length < 50) 
    return { signal: 'NONE', reason: 'Insufficient 5m data' }

  // Try BUY first
  const buyC1 = score1hTrend(candles1h, 'BUY')
  
  if (buyC1 >= 6) {
    // 1h supports BUY, check 15m
    const buyC2 = score15mConfirmation(candles15m, 'BUY')
    
    if (buyC2 >= 6) {
      // Both 1h and 15m support BUY, check entry
      const buyC3 = score5mEntry(candles5m, 'BUY')
      const buyTotal = buyC1 + buyC2 + buyC3
      
      if (buyTotal >= 20) {
        return {
          signal: 'BUY',
          total: buyTotal,
          c1: buyC1,
          c2: buyC2,
          c3: buyC3,
          reason: 'Three timeframe bull alignment'
        }
      }
    }
  }

  // Try SELL
  const sellC1 = score1hTrend(candles1h, 'SELL')
  
  if (sellC1 >= 6) {
    const sellC2 = score15mConfirmation(candles15m, 'SELL')
    
    if (sellC2 >= 6) {
      const sellC3 = score5mEntry(candles5m, 'SELL')
      const sellTotal = sellC1 + sellC2 + sellC3
      
      if (sellTotal >= 20) {
        return {
          signal: 'SELL',
          total: sellTotal,
          c1: sellC1,
          c2: sellC2,
          c3: sellC3,
          reason: 'Three timeframe bear alignment'
        }
      }
    }
  }

  return { 
    signal: 'NONE', 
    reason: 'No three timeframe alignment',
    buyC1, sellC1
  }
}
