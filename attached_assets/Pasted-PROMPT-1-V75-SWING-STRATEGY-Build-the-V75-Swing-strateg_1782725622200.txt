function build4hCandles(candles1h: Candle[]): Candle[] {
  const grouped: { [key: number]: Candle[] } = {}
  
  candles1h.forEach(candle => {
    const boundary = Math.floor(candle.time / 14400) * 14400
    if (!grouped[boundary]) grouped[boundary] = []
    grouped[boundary].push(candle)
  })
  
  return Object.entries(grouped)
    .filter(([_, g]) => g.length > 0)
    .map(([boundary, group]) => {
      const sorted = group.sort((a,b) => a.time - b.time)
      return {
        time: parseInt(boundary),
        open: sorted[0].open,
        high: Math.max(...sorted.map(c => c.high)),
        low: Math.min(...sorted.map(c => c.low)),
        close: sorted[sorted.length-1].close
      }
    })
    .sort((a,b) => a.time - b.time)
}

PART 3 — CONSOLIDATION RANGE DETECTOR
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

The swing strategy only enters after a 
consolidation range breaks out and retests.
This detector identifies active ranges.

function detectConsolidationRange(
  candles1h: Candle[]
): ConsolidationRange | null {

  const last20 = candles1h.slice(-20)
  
  // Calculate how tight the range is
  const highs = last20.map(c => c.high)
  const lows = last20.map(c => c.low)
  const rangeHigh = Math.max(...highs)
  const rangeLow = Math.min(...lows)
  const rangeSize = rangeHigh - rangeLow
  const midpoint = (rangeHigh + rangeLow) / 2
  const rangePct = rangeSize / midpoint * 100

  // Count candles within the range boundaries
  const candlesInRange = last20.filter(c =>
    c.high <= rangeHigh * 1.001 &&
    c.low >= rangeLow * 0.999
  ).length

  // Consolidation confirmed when:
  // Range is less than 1.5% of price
  // At least 12 of 20 candles are within range
  const isConsolidating = rangePct < 1.5 && 
                          candlesInRange >= 12

  if (!isConsolidating) return null

  // Calculate range duration
  const duration = last20.length

  console.log(`Consolidation detected:
    High: ${rangeHigh}
    Low: ${rangeLow}  
    Size: ${rangeSize.toFixed(2)} pips
    Pct: ${rangePct.toFixed(3)}%
    Candles in range: ${candlesInRange}/20`)

  return {
    high: rangeHigh,
    low: rangeLow,
    midpoint,
    size: rangeSize,
    duration,
    confirmed: true
  }
}

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

function detectBreakout(
  candles1h: Candle[],
  range: ConsolidationRange
): BreakoutResult | null {

  const lastCandle = candles1h[candles1h.length - 1]
  const prevCandle = candles1h[candles1h.length - 2]

  // Bullish breakout: close above range high
  // with body (not just wick) above range high
  const bullBreakout = 
    lastCandle.close > range.high &&
    lastCandle.open > range.high * 0.998 &&
    lastCandle.close > prevCandle.close &&
    // Breakout candle must be larger than average
    (lastCandle.high - lastCandle.low) > 
    (range.size * 0.3)

  // Bearish breakout: close below range low
  const bearBreakout =
    lastCandle.close < range.low &&
    lastCandle.open < range.low * 1.002 &&
    lastCandle.close < prevCandle.close &&
    (lastCandle.high - lastCandle.low) >
    (range.size * 0.3)

  if (bullBreakout) {
    console.log(`BULL BREAKOUT detected at 
      ${lastCandle.close} above range high 
      ${range.high}`)
    return {
      direction: 'BUY',
      breakoutPrice: lastCandle.close,
      retestLevel: range.high,
      stopLevel: range.low,
      breakoutCandle: lastCandle
    }
  }

  if (bearBreakout) {
    console.log(`BEAR BREAKOUT detected at 
      ${lastCandle.close} below range low 
      ${range.low}`)
    return {
      direction: 'SELL',
      breakoutPrice: lastCandle.close,
      retestLevel: range.low,
      stopLevel: range.high,
      breakoutCandle: lastCandle
    }
  }

  return null
}

PART 5 — RETEST DETECTOR
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

After breakout, wait for price to pull back
and retest the broken level before entering.

function detectRetest(
  currentPrice: number,
  breakout: BreakoutResult,
  candles15m: Candle[]
): RetestResult {

  const retestLevel = breakout.retestLevel
  const direction = breakout.direction

  // How close is price to the retest level?
  const distanceFromLevel = Math.abs(
    currentPrice - retestLevel
  )
  const distancePct = distanceFromLevel / 
    retestLevel * 100

  // Price is at retest level when within 0.1%
  const atRetestLevel = distancePct < 0.1

  // Check for rejection from retest level on 15m
  const last3candles15m = candles15m.slice(-3)
  let rejectionConfirmed = false

  if (direction === 'BUY' && atRetestLevel) {
    // For buy retest: price should touch level
    // then close ABOVE it on 15m
    const lastCandle = last3candles15m[
      last3candles15m.length - 1
    ]
    rejectionConfirmed = 
      lastCandle.low <= retestLevel * 1.001 &&
      lastCandle.close > retestLevel &&
      lastCandle.close > lastCandle.open
      // Bullish close after touching level
  }

  if (direction === 'SELL' && atRetestLevel) {
    const lastCandle = last3candles15m[
      last3candles15m.length - 1
    ]
    rejectionConfirmed =
      lastCandle.high >= retestLevel * 0.999 &&
      lastCandle.close < retestLevel &&
      lastCandle.close < lastCandle.open
      // Bearish close after touching level
  }

  return {
    atLevel: atRetestLevel,
    rejected: rejectionConfirmed,
    distancePct,
    readyToEnter: atRetestLevel && rejectionConfirmed
  }
}

function score4hTrend(candles4h, direction):
  closes = candles4h.map(c => c.close)
  ema20_4h = calculateEMA(closes, 20)
  ema50_4h = calculateEMA(closes, 50)
  currentPrice = closes[closes.length - 1]
  
  // Calculate slope of EMA20
  ema20_3bars_ago = calculateEMA(
    closes.slice(0, -3), 20
  )
  ema20Slope = ema20_4h - ema20_3bars_ago
  
  if direction === 'BUY':
    if ema20_4h > ema50_4h AND 
       currentPrice > ema20_4h AND
       ema20Slope > 0:
      return 10 // Perfect uptrend
    if ema20_4h > ema50_4h AND
       currentPrice > ema50_4h:
      return 7  // Uptrend, price slightly below EMA20
    if ema20_4h > ema50_4h:
      return 4  // Weak uptrend signal
    return 0    // Downtrend — no BUY
  
  if direction === 'SELL':
    if ema20_4h < ema50_4h AND
       currentPrice < ema20_4h AND
       ema20Slope < 0:
      return 10
    if ema20_4h < ema50_4h AND
       currentPrice < ema50_4h:
      return 7
    if ema20_4h < ema50_4h:
      return 4
    return 0

COMPONENT 3 — BREAKOUT QUALITY (0-5)
Scores the quality of the initial breakout.

function scoreBreakoutQuality(
  breakout: BreakoutResult,
  range: ConsolidationRange
): number {

  // How decisive was the breakout candle?
  const breakoutCandle = breakout.breakoutCandle
  const candleBody = Math.abs(
    breakoutCandle.close - breakoutCandle.open
  )
  const candleRange = 
    breakoutCandle.high - breakoutCandle.low
  const bodyRatio = candleBody / candleRange

  // Strong body = decisive breakout
  if bodyRatio > 0.7 AND 
     candleRange > range.size * 0.5:
    return 5 // Strong decisive breakout
  if bodyRatio > 0.5:
    return 3 // Moderate breakout
  if bodyRatio > 0.3:
    return 2 // Weak breakout
  return 1   // Very weak — wick dominated
}
