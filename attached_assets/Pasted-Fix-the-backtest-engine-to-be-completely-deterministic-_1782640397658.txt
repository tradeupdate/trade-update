async function getHistoricalCandles(
  symbol: string,
  granularity: number,
  dateFrom: number,
  dateTo: number
): Promise<Candle[]> {

  const cacheKey = `${symbol}_${granularity}m_${dateFrom}_${dateTo}`
  const cacheFile = path.join(
    __dirname, 
    '../data/candle-cache',
    `${cacheKey}.json`
  )

  // CHECK CACHE FIRST
  if (fs.existsSync(cacheFile)) {
    console.log(`Cache hit: ${cacheKey}`)
    const cached = JSON.parse(fs.readFileSync(cacheFile, 'utf8'))
    console.log(`Loaded ${cached.candles.length} cached candles`)
    console.log(`Cache created: ${new Date(cached.cachedAt).toISOString()}`)
    return cached.candles
  }

  console.log(`Cache miss: fetching from Deriv...`)
  
  // FETCH FROM DERIV
  const candles = await fetchFromDeriv(symbol, granularity, dateFrom, dateTo)
  
  // SAVE TO CACHE
  const cacheData = {
    symbol,
    granularity,
    dateFrom,
    dateTo,
    candles,
    cachedAt: Date.now(),
    count: candles.length
  }
  
  fs.mkdirSync(path.dirname(cacheFile), { recursive: true })
  fs.writeFileSync(cacheFile, JSON.stringify(cacheData, null, 2))
  console.log(`Cached ${candles.length} candles to ${cacheFile}`)
  
  return candles
}

function normalizeCandles(rawCandles: any[]): Candle[] {
  
  // 1. Map to standard format
  const candles: Candle[] = rawCandles.map(c => ({
    time: typeof c.epoch === 'number' ? c.epoch :
          typeof c.time === 'number' ? c.time :
          parseInt(c.epoch || c.time),
    open: parseFloat(c.open),
    high: parseFloat(c.high),
    low: parseFloat(c.low),
    close: parseFloat(c.close)
  }))
  
  // 2. Remove any with invalid data
  const valid = candles.filter(c => 
    !isNaN(c.time) &&
    !isNaN(c.open) &&
    !isNaN(c.high) &&
    !isNaN(c.low) &&
    !isNaN(c.close) &&
    c.high >= c.low &&
    c.high >= c.open &&
    c.high >= c.close &&
    c.low <= c.open &&
    c.low <= c.close &&
    c.close > 0
  )
  
  // 3. Sort by time ASCENDING (oldest first)
  const sorted = valid.sort((a, b) => a.time - b.time)
  
  // 4. Remove duplicates (keep first occurrence)
  const seen = new Set<number>()
  const deduped = sorted.filter(c => {
    if (seen.has(c.time)) return false
    seen.add(c.time)
    return true
  })
  
  // 5. Validate price range for R_75
  const prices = deduped.map(c => c.close)
  const minPrice = Math.min(...prices)
  const maxPrice = Math.max(...prices)
  
  console.log(`Candle validation:
    Raw: ${rawCandles.length}
    Valid: ${valid.length}
    After dedup: ${deduped.length}
    Price range: ${minPrice.toFixed(2)} - ${maxPrice.toFixed(2)}
    Time range: ${new Date(deduped[0].time * 1000).toISOString()} 
              to ${new Date(deduped[deduped.length-1].time * 1000).toISOString()}
  `)
  
  if (minPrice < 20000 || maxPrice > 80000) {
    console.error('PRICE RANGE ERROR: Prices outside R_75 expected range')
  }
  
  return deduped
}

Building 5m and 15m from 1m candles must be

function build5mCandles(candles1m: Candle[]): Candle[] {
  const grouped: { [key: number]: Candle[] } = {}
  
  candles1m.forEach(candle => {
    // Round down to nearest 5-minute boundary
    // This is DETERMINISTIC - same candle always
    // goes to same group
    const boundary = Math.floor(candle.time / 300) * 300
    if (!grouped[boundary]) grouped[boundary] = []
    grouped[boundary].push(candle)
  })
  
  const candles5m: Candle[] = Object.entries(grouped)
    .filter(([_, group]) => group.length > 0)
    .map(([boundary, group]) => {
      // Sort group by time to ensure consistent OHLC
      const sorted = group.sort((a, b) => a.time - b.time)
      return {
        time: parseInt(boundary),
        open: sorted[0].open,
        high: Math.max(...sorted.map(c => c.high)),
        low: Math.min(...sorted.map(c => c.low)),
        close: sorted[sorted.length - 1].close
      }
    })
    .sort((a, b) => a.time - b.time)
  
  return candles5m
}

function build15mCandles(candles1m: Candle[]): Candle[] {
  const grouped: { [key: number]: Candle[] } = {}
  
  candles1m.forEach(candle => {
    // Round down to nearest 15-minute boundary
    const boundary = Math.floor(candle.time / 900) * 900
    if (!grouped[boundary]) grouped[boundary] = []
    grouped[boundary].push(candle)
  })
  
  const candles15m: Candle[] = Object.entries(grouped)
    .filter(([_, group]) => group.length > 0)
    .map(([boundary, group]) => {
      const sorted = group.sort((a, b) => a.time - b.time)
      return {
        time: parseInt(boundary),
        open: sorted[0].open,
        high: Math.max(...sorted.map(c => c.high)),
        low: Math.min(...sorted.map(c => c.low)),
        close: sorted[sorted.length - 1].close
      }
    })
    .sort((a, b) => a.time - b.time)
  
  return candles15m
}

function candleChecksum(candles: Candle[]): string {
  const sum = candles.reduce((acc, c) => acc + c.close, 0)
  const count = candles.length
  return `${count}:${sum.toFixed(2)}`
}
