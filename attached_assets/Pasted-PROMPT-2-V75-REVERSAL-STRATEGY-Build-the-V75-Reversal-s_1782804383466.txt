function calculateSessionMove(
  candles5m: Candle[],
  currentTime: number
): SessionMoveResult {

  // Get session open time
  const utcHour = new Date(
    currentTime * 1000
  ).getUTCHours()
  
  let sessionOpenHour: number
  if (utcHour >= 0 && utcHour < 7) 
    sessionOpenHour = 0   // Asian
  else if (utcHour >= 7 && utcHour < 12) 
    sessionOpenHour = 7   // London
  else if (utcHour >= 12 && utcHour < 17) 
    sessionOpenHour = 12  // Overlap/NY
  else 
    sessionOpenHour = 0   // Default to Asian

  // Find session open candle
  const sessionOpenTime = new Date(
    currentTime * 1000
  )
  sessionOpenTime.setUTCHours(sessionOpenHour, 0, 0, 0)
  const sessionOpenUnix = sessionOpenTime.getTime() / 1000

  // Find the first candle of this session
  const sessionCandles = candles5m.filter(
    c => c.time >= sessionOpenUnix && 
         c.time <= currentTime
  )

  if (sessionCandles.length === 0) {
    return { 
      sessionOpenPrice: 0, 
      currentMove: 0, 
      moveDirection: 'NONE',
      sufficientMove: false 
    }
  }

  const sessionOpenPrice = sessionCandles[0].open
  const currentPrice = 
    sessionCandles[sessionCandles.length-1].close
  const rawMove = currentPrice - sessionOpenPrice
  const moveInPips = Math.abs(rawMove)
  const moveDirection = rawMove > 0 ? 'UP' : 'DOWN'

  console.log(`Session move from open:
    Session open: ${sessionOpenPrice}
    Current: ${currentPrice}
    Move: ${moveInPips.toFixed(2)} pips 
    Direction: ${moveDirection}
    Sufficient (350+): ${moveInPips >= 350}`)

  return {
    sessionOpenPrice,
    currentMove: moveInPips,
    moveDirection,
    sufficientMove: moveInPips >= 350
  }
}

If sessionMove.sufficientMove === false:

  ${sessionMove.currentMove.toFixed(0)} pips

PART 3 — DUAL TIMEFRAME DIVERGENCE DETECTOR
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Both 5m AND 15m must show divergence.

function detectDivergence(
  candles: Candle[],
  lookback: number = 10
): DivergenceResult {

  const recent = candles.slice(-lookback)
  const closes = recent.map(c => c.close)
  const rsiValues = calculateRSIArray(recent, 14)
  // RSI for each candle not just current

  // Find price swing highs and lows
  let priceHighIdx = 0
  let priceLowIdx = 0
  let rsiHighIdx = 0
  let rsiLowIdx = 0

  for (let i = 1; i < recent.length - 1; i++) {
    if (closes[i] > closes[priceHighIdx])
      priceHighIdx = i
    if (closes[i] < closes[priceLowIdx])
      priceLowIdx = i
    if (rsiValues[i] > rsiValues[rsiHighIdx])
      rsiHighIdx = i
    if (rsiValues[i] < rsiValues[rsiLowIdx])
      rsiLowIdx = i
  }

  const lastPriceHigh = closes[priceHighIdx]
  const prevPriceHigh = Math.max(
    ...closes.slice(0, priceHighIdx)
  )
  const lastRsiHigh = rsiValues[rsiHighIdx]
  const prevRsiHigh = Math.max(
    ...rsiValues.slice(0, rsiHighIdx)
  )

  const lastPriceLow = closes[priceLowIdx]
  const prevPriceLow = Math.min(
    ...closes.slice(0, priceLowIdx)
  )
  const lastRsiLow = rsiValues[rsiLowIdx]
  const prevRsiLow = Math.min(
    ...rsiValues.slice(0, rsiLowIdx)
  )

  // Bearish divergence: price makes higher high
  // but RSI makes lower high
  const bearishDivergence =
    lastPriceHigh > prevPriceHigh &&
    lastRsiHigh < prevRsiHigh &&
    rsiValues[rsiValues.length-1] > 65
    // Current RSI still elevated

  // Bullish divergence: price makes lower low
  // but RSI makes higher low
  const bullishDivergence =
    lastPriceLow < prevPriceLow &&
    lastRsiLow > prevRsiLow &&
    rsiValues[rsiValues.length-1] < 35
    // Current RSI still depressed

  return {
    bullish: bullishDivergence,
    bearish: bearishDivergence,
    currentRsi: rsiValues[rsiValues.length-1]
  }
}

function detectBBExtreme(
  candles5m: Candle[]
): BBExtremeResult {

  const bb = calculateBB(candles5m, 20, 2.0)
  const bb25 = calculateBB(candles5m, 20, 2.5)
  const currentPrice = 
    candles5m[candles5m.length-1].close

  const aboveUpper = currentPrice > bb.upper
  const belowLower = currentPrice < bb.lower
  const above25 = currentPrice > bb25.upper
  const below25 = currentPrice < bb25.lower

  return {
    aboveUpper,   // Standard 2σ breach
    belowLower,
    above25sigma: above25,  // Strong 2.5σ breach
    below25sigma: below25,
    // Premium signal = divergence + 2.5σ breach
    premiumSell: above25,
    premiumBuy: below25,
    middle: bb.middle
  }
}

PART 5 — 1M CONFIRMATION CANDLE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

After divergence and BB extreme confirmed,
wait for 1m candle confirmation.

function check1mConfirmation(
  candles1m: Candle[],
  direction: 'BUY' | 'SELL'
): boolean {

  const last = candles1m[candles1m.length-1]
  const prev = candles1m[candles1m.length-2]

  const candleBody = Math.abs(last.close - last.open)
  const candleRange = last.high - last.low
  const bodyRatio = candleBody / candleRange

  if (direction === 'BUY') {
    // Confirmation: green candle
    const isGreen = last.close > last.open
    // Closes above midpoint of previous candle
    const abovePrevMid = last.close > 
      (prev.high + prev.low) / 2
    // Body dominated (not wick dominated)
    const bodyDominated = bodyRatio > 0.5
    // Wick in trend direction not too large
    const lowerWick = last.open - last.low
    const upperBody = last.close - last.open
    const wickNotTooLarge = lowerWick < upperBody * 2

    const confirmed = isGreen && abovePrevMid && 
      bodyDominated && wickNotTooLarge

    console.log(`1m BUY confirmation:
      Green: ${isGreen}
      Above prev mid: ${abovePrevMid}
      Body dominated: ${bodyDominated}
      Wick ok: ${wickNotTooLarge}
      CONFIRMED: ${confirmed}`)

    return confirmed
  }

  if (direction === 'SELL') {
    const isRed = last.close < last.open
    const belowPrevMid = last.close < 
      (prev.high + prev.low) / 2
    const bodyDominated = bodyRatio > 0.5
    const upperWick = last.high - last.open
    const lowerBody = last.open - last.close
    const wickNotTooLarge = upperWick < lowerBody * 2

    return isRed && belowPrevMid && 
      bodyDominated && wickNotTooLarge
  }

  return false
}

COMPONENT 1 — SESSION EXHAUSTION (0-10)
How far has price moved from session open?

function scoreSessionExhaustion(
  sessionMove: SessionMoveResult,
  direction: 'BUY' | 'SELL'
): number {

  const move = sessionMove.currentMove
  const moveDir = sessionMove.moveDirection

  // Reversal direction must oppose session move
  if (direction === 'BUY' && moveDir !== 'DOWN')
    return 0 // Session moved UP, not down
  if (direction === 'SELL' && moveDir !== 'UP')
    return 0 // Session moved DOWN, not up

  if (move >= 600) return 10 // Extreme exhaustion
  if (move >= 500) return 9
  if (move >= 450) return 8
  if (move >= 400) return 7
  if (move >= 375) return 6
  if (move >= 350) return 5 // Minimum threshold
  return 0
}

COMPONENT 2 — DIVERGENCE STRENGTH (0-10)
How strong is the divergence signal?

function scoreDivergenceStrength(
  div5m: DivergenceResult,
  div15m: DivergenceResult,
  direction: 'BUY' | 'SELL'
): number {

  let score = 0

  if (direction === 'BUY') {
    if (div5m.bullish) score += 4
    if (div15m.bullish) score += 4
    // RSI extreme bonus
    if (div5m.currentRsi < 25) score += 2
    else if (div5m.currentRsi < 30) score += 1
  }

  if (direction === 'SELL') {
    if (div5m.bearish) score += 4
    if (div15m.bearish) score += 4
    if (div5m.currentRsi > 75) score += 2
    else if (div5m.currentRsi > 70) score += 1
  }

  return Math.min(score, 10)
}

COMPONENT 3 — BB BREACH QUALITY (0-5)
How far beyond the bands is price?

function scoreBBBreach(
  bbExtreme: BBExtremeResult,
  direction: 'BUY' | 'SELL'
): number {

  if (direction === 'BUY') {
    if (bbExtreme.below25sigma) return 5
    // 2.5σ breach = maximum score
    if (bbExtreme.belowLower) return 3
    // 2σ breach = moderate score
    return 1 // Near lower band
  }

  if (direction === 'SELL') {
    if (bbExtreme.above25sigma) return 5
    if (bbExtreme.aboveUpper) return 3
    return 1
  }

  return 0
}
