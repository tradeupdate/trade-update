Replace hardcoded 0.85 payout assumption

async function getActualPayoutRate(
  derivService: DerivService,
  pair: string,
  stake: number
): Promise<number> {

  try {
    const proposal = await derivService.send({
      proposal: 1,
      amount: stake,
      basis: "stake",
      contract_type: "CALL",
      currency: "USD",
      duration: 5,
      duration_unit: "m",
      symbol: pair
    })

    const payout = proposal.proposal.payout
    const askPrice = proposal.proposal.ask_price
    const payoutRate = (payout - askPrice) / askPrice

    console.log(`Actual payout rate for ${pair}:
      Stake: $${stake}
      Payout: $${payout}
      Ask price: $${askPrice}
      Rate: ${(payoutRate*100).toFixed(2)}%`)

    return payoutRate

  } catch (error) {
    console.error(`Failed to get payout rate, 
      using default 0.85:`, error)
    return 0.85 // Fallback
  }
}

Cache the payout rate per pair, refresh every
30 minutes since it can drift slightly:

const payoutCache: { [pair: string]: 
  { rate: number, updatedAt: number } } = {}

async function getCachedPayoutRate(
  derivService: DerivService,
  pair: string,
  stake: number
): Promise<number> {

  const cached = payoutCache[pair]
  const CACHE_DURATION = 30 * 60 * 1000

  if (cached && 
      Date.now() - cached.updatedAt < CACHE_DURATION) {
    return cached.rate
  }

  const rate = await getActualPayoutRate(
    derivService, pair, stake
  )
  
  payoutCache[pair] = { 
    rate, 
    updatedAt: Date.now() 
  }

  return rate
}

function detectRangeCleanliness(
  candles5m: Candle[]
): RangeQualityResult {

  const last30 = candles5m.slice(-30)
  
  // Calculate the range
  const highs = last30.map(c => c.high)
  const lows = last30.map(c => c.low)
  const rangeHigh = Math.max(...highs)
  const rangeLow = Math.min(...lows)
  const rangeSize = rangeHigh - rangeLow
  const midpoint = (rangeHigh + rangeLow) / 2

  // Count how many candles respected the range
  // (didn't break out significantly)
  const candlesInRange = last30.filter(c =>
    c.high <= rangeHigh * 1.002 &&
    c.low >= rangeLow * 0.998
  ).length

  const rangeRespect = candlesInRange / 30

  // Calculate noise — how choppy is price action
  // within the range (lots of small wicks = noisy)
  let totalWickSize = 0
  let totalBodySize = 0
  
  last30.forEach(c => {
    const body = Math.abs(c.close - c.open)
    const wick = (c.high - c.low) - body
    totalWickSize += wick
    totalBodySize += body
  })

  const wickToBodyRatio = totalWickSize / 
    (totalBodySize || 1)

  // Calculate how many times price touched 
  // the band edges (more touches = cleaner range)
  const bb = calculateBB(last30, 20, 2.0)
  const upperTouches = last30.filter(c => 
    c.high >= bb.upper * 0.998
  ).length
  const lowerTouches = last30.filter(c =>
    c.low <= bb.lower * 1.002
  ).length

  // CLEANLINESS SCORE (0-10)
  let cleanlinessScore = 0

  if (rangeRespect >= 0.85) cleanlinessScore += 4
  else if (rangeRespect >= 0.70) cleanlinessScore += 2

  if (wickToBodyRatio < 1.5) cleanlinessScore += 3
  else if (wickToBodyRatio < 2.5) cleanlinessScore += 1

  if (upperTouches >= 1 && lowerTouches >= 1) 
    cleanlinessScore += 3
  else if (upperTouches >= 1 || lowerTouches >= 1)
    cleanlinessScore += 1

  console.log(`Range cleanliness check:
    Range respect: ${(rangeRespect*100).toFixed(0)}%
    Wick/body ratio: ${wickToBodyRatio.toFixed(2)}
    Upper touches: ${upperTouches}
    Lower touches: ${lowerTouches}
    Cleanliness score: ${cleanlinessScore}/10`)

  return {
    cleanlinessScore,
    rangeHigh,
    rangeLow,
    midpoint,
    tradeable: cleanlinessScore >= 6
  }
}

function detectV10TrendRisk(
  candles15m: Candle[]
): TrendRiskResult {

  const adx = calculateADX(candles15m, 14)
  const last10 = candles15m.slice(-10)
  
  // Directional consistency check
  const bullish = last10.filter(
    c => c.close > c.open
  ).length
  const bearish = 10 - bullish

  const strongDirectional = 
    bullish >= 8 || bearish >= 8

  const trendRisk = adx > 25 || strongDirectional

  console.log(`V10 trend risk check:
    ADX: ${adx.toFixed(1)}
    Directional candles: ${Math.max(bullish, bearish)}/10
    TREND RISK: ${trendRisk}`)

  return {
    adx,
    trendRisk,
    safe: adx < 20 && !strongDirectional
  }
}

COMPONENT 2 — BB EXTREME + RSI CONFLUENCE (0-10)

function scoreV10Entry(
  candles5m: Candle[],
  direction: 'BUY' | 'SELL'
): number {

  const bb = calculateBB(candles5m, 20, 2.0)
  const rsi = calculateRSI(candles5m, 14)
  const currentPrice = 
    candles5m[candles5m.length-1].close
  const stoch = calculateStochastic(candles5m, 5, 3, 3)

  let score = 0

  if (direction === 'BUY') {
    // Price at or below lower band
    const atLowerBand = currentPrice <= bb.lower * 1.001
    const belowLowerBand = currentPrice < bb.lower

    if (belowLowerBand) score += 4
    else if (atLowerBand) score += 3

    // RSI oversold
    if (rsi < 25) score += 3
    else if (rsi < 30) score += 2
    else if (rsi < 35) score += 1

    // Stochastic confirmation
    if (stoch.k < 20 && stoch.k > stoch.d) score += 3
    else if (stoch.k < 25) score += 1
  }

  if (direction === 'SELL') {
    const atUpperBand = currentPrice >= bb.upper * 0.999
    const aboveUpperBand = currentPrice > bb.upper

    if (aboveUpperBand) score += 4
    else if (atUpperBand) score += 3

    if (rsi > 75) score += 3
    else if (rsi > 70) score += 2
    else if (rsi > 65) score += 1

    if (stoch.k > 80 && stoch.k < stoch.d) score += 3
    else if (stoch.k > 75) score += 1
  }

  return Math.min(score, 10)
}

COMPONENT 3 — REJECTION CONFIRMATION (0-5)

function scoreRejection(
  candles1m: Candle[],
  direction: 'BUY' | 'SELL'
): number {

  const last = candles1m[candles1m.length-1]
  const body = Math.abs(last.close - last.open)
  const range = last.high - last.low

  if (direction === 'BUY') {
    const lowerWick = last.open < last.close ?
      last.open - last.low : last.close - last.low
    const wickRatio = lowerWick / (range || 1)

    if (wickRatio > 0.6 && last.close > last.open) 
      return 5 // Strong rejection wick + green close
    if (wickRatio > 0.4) return 3
    if (last.close > last.open) return 2
    return 0
  }

  if (direction === 'SELL') {
    const upperWick = last.close < last.open ?
      last.open - last.high : last.close - last.high
    const wickRatio = Math.abs(upperWick) / (range || 1)

    if (wickRatio > 0.6 && last.close < last.open)
      return 5
    if (wickRatio > 0.4) return 3
    if (last.close < last.open) return 2
    return 0
  }

  return 0
}

PART 8 — SIGNAL GENERATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

function generateV10Signal(
  candles1m, candles5m, candles15m
): SignalResult {

  // Step 1: Trend risk check (safety first)
  const trendRisk = detectV10TrendRisk(candles15m)
  if (trendRisk.trendRisk) {
    return { 
      signal: 'NONE', 
      reason: 'V10 trending — mean reversion paused' 
    }
  }

  // Step 2: Range cleanliness check
  const rangeQuality = detectRangeCleanliness(candles5m)
  if (!rangeQuality.tradeable) {
    return { 
      signal: 'NONE', 
      reason: `Range too noisy: 
        ${rangeQuality.cleanlinessScore}/10` 
    }
  }

  const c1 = rangeQuality.cleanlinessScore

  // Step 3: Try BUY (price at lower band)
  const buyC2 = scoreV10Entry(candles5m, 'BUY')
  if (buyC2 >= 6) {
    const buyC3 = scoreRejection(candles1m, 'BUY')
    const buyTotal = c1 + buyC2 + buyC3

    if (buyTotal >= 18) {
      return {
        signal: 'BUY',
        total: buyTotal,
        c1, c2: buyC2, c3: buyC3,
        reason: 'Range extreme + RSI oversold + rejection'
      }
    }
  }

  // Step 4: Try SELL (price at upper band)
  const sellC2 = scoreV10Entry(candles5m, 'SELL')
  if (sellC2 >= 6) {
    const sellC3 = scoreRejection(candles1m, 'SELL')
    const sellTotal = c1 + sellC2 + sellC3

    if (sellTotal >= 18) {
      return {
        signal: 'SELL',
        total: sellTotal,
        c1, c2: sellC2, c3: sellC3,
        reason: 'Range extreme + RSI overbought + rejection'
      }
    }
  }

  return { signal: 'NONE', reason: 'No qualifying setup' }
}
