function getSimpleSwingSignal(

    Math.min(50, closes4h.length))

  const bear4h = ema20_4h < ema50_4h

    Math.min(9, closes1h.length))

  const bear1h = ema9_1h < ema21_1h

    Math.min(21, closes5m.length))
  const rsi5m = calculateRSI(candles5m, 14)

  const nearEMA = Math.abs(price - ema21_5m) / 
    ema21_5m < 0.005 // Within 0.5%

  // LOG EVERY 500 CANDLES

      5m: EMA21=${ema21_5m?.toFixed(0)}
          RSI=${rsi5m?.toFixed(1)}
          NearEMA=${nearEMA}`)
  }

  // BUY SIGNAL
  if (bull4h && bull1h && nearEMA && 
      rsi5m > 30 && rsi5m < 55) {
    console.log(`SWING BUY SIGNAL at ${

function executeSwingTrade(
  state: any,
  direction: string,
  candle: Candle,
  idx: number
): void {

  const entryPrice = candle.close
  const atr = 100 // Default ATR if calc fails
  const stopDistance = Math.max(80, atr * 2)

  let stopLoss: number
  let takeProfit: number

  if (direction === 'BUY') {
    stopLoss = entryPrice - stopDistance
    takeProfit = entryPrice + (stopDistance * 2.0)
  } else {
    stopLoss = entryPrice + stopDistance
    takeProfit = entryPrice - (stopDistance * 2.0)
  }

  const stake = Math.max(
    1, Math.min(50, state.balance * 0.01)
  )

  state.openTrade = {
    direction,
    entryPrice,
    stopLoss,
    takeProfit,
    stake,
    openIdx: idx,
    openTime: candle.time
  }

  state.todayTrades++

  console.log(`>>> SWING TRADE OPENED:
    Direction: ${direction}
    Entry: ${entryPrice}
    Stop: ${stopLoss}
    TP: ${takeProfit}
    Stake: $${stake}
    Balance: $${state.balance}`)
}

function checkSwingOutcome(
  trade: any,
  candle: Candle
): string | null {

  if (trade.direction === 'BUY') {
    if (candle.low <= trade.stopLoss) 
      return 'STOP_LOSS'
    if (candle.high >= trade.takeProfit) 
      return 'TP_HIT'
  } else {
    if (candle.high >= trade.stopLoss) 
      return 'STOP_LOSS'
    if (candle.low <= trade.takeProfit) 
      return 'TP_HIT'
  }
  return null
}

function closeSwingTrade(
  state: any,
  outcome: string,
  candle: Candle
): void {

  const trade = state.openTrade
  let pnl: number
  let closePrice: number

  if (outcome === 'TP_HIT') {
    closePrice = trade.takeProfit
    pnl = trade.stake * 0.85
  } else if (outcome === 'STOP_LOSS') {
    closePrice = trade.stopLoss
    pnl = -(trade.stake)
  } else {
    closePrice = candle.close
    const moved = trade.direction === 'BUY' ?
      closePrice - trade.entryPrice :
      trade.entryPrice - closePrice
    pnl = moved > 0 ? 
      trade.stake * 0.4 : -(trade.stake * 0.4)
  }

  state.balance += pnl

  if (pnl > 0) state.consecutiveLosses = 0
  else state.consecutiveLosses++

  const closedTrade = {
    ...trade,
    exitPrice: closePrice,
    pnl,
    closeReason: outcome,
    closeTime: candle.time,
    balanceAfter: state.balance
  }

  state.allTrades.push(closedTrade)
  state.openTrade = null

  console.log(`<<< SWING TRADE CLOSED:
    Direction: ${trade.direction}
    Entry: ${trade.entryPrice}
    Exit: ${closePrice}
    P&L: $${pnl.toFixed(2)}
    Reason: ${outcome}
    Balance: $${state.balance.toFixed(2)}`)
}

function calculateSwingResults(state: any) {

  const wins = state.allTrades.filter(
    (t:any) => t.pnl > 0
  ).length
  const totalPnl = state.allTrades.reduce(
    (s:number, t:any) => s + t.pnl, 0
  )

  console.log(`=== SWING BACKTEST COMPLETE ===
    Total trades: ${total}
    Wins: ${wins}
    Win rate: ${total > 0 ? 
      (wins/total*100).toFixed(1) : 0}%
    Total P&L: $${totalPnl.toFixed(2)}`)

  return {
    totalTrades: total,
    wins,
    losses: total - wins,
    winRate: total > 0 ? wins/total*100 : 0,
    totalPnl,
    finalBalance: state.balance,
    trades: state.allTrades
  }
}

// At server module level (not inside a function)
// This persists for the server lifetime:

const GLOBAL_CANDLE_CACHE: Map<string, {
  candles: Candle[]
  fetchedAt: number
  hash: string
}> = new Map()

function getCacheKey(
  symbol: string,
  dateFrom: number | string,
  dateTo: number | string
): string {
  // Always normalize to UTC midnight
  const fromDate = new Date(
    typeof dateFrom === 'string' ? 
    dateFrom : dateFrom * 1000
  )
  fromDate.setUTCHours(0, 0, 0, 0)
  
  const toDate = new Date(
    typeof dateTo === 'string' ?
    dateTo : dateTo * 1000
  )
  toDate.setUTCHours(23, 59, 59, 0)
  
  const fromUnix = Math.floor(
    fromDate.getTime() / 1000
  )
  const toUnix = Math.floor(
    toDate.getTime() / 1000
  )
  
  return `${symbol}_5m_${fromUnix}_${toUnix}`
}

async function getBacktestCandles(
  symbol: string,
  dateFrom: number | string,
  dateTo: number | string
): Promise<Candle[]> {

  const key = getCacheKey(symbol, dateFrom, dateTo)

  // Check memory cache first
  if (GLOBAL_CANDLE_CACHE.has(key)) {
    const cached = GLOBAL_CANDLE_CACHE.get(key)!
    console.log(`Memory cache hit: ${key}
      Candles: ${cached.candles.length}
      Hash: ${cached.hash}`)
    return cached.candles
  }

  // Check disk cache
  const cacheFile = path.join(
    __dirname, '../data/candle-cache',
    `${key}.json`
  )

  if (fs.existsSync(cacheFile)) {
    const disk = JSON.parse(
      fs.readFileSync(cacheFile, 'utf8')
    )
    const candles = disk.candles as Candle[]
    const hash = computeHash(candles)
    
    // Store in memory too
    GLOBAL_CANDLE_CACHE.set(key, {
      candles, fetchedAt: disk.fetchedAt, hash
    })
    
    console.log(`Disk cache hit: ${key}
      Candles: ${candles.length}
      Hash: ${hash}`)
    return candles
  }

  // Fetch from Deriv
  console.log(`Cache miss: fetching ${key}`)
  const candles = await fetchBacktestCandles(
    symbol,
    typeof dateFrom === 'string' ?
      Math.floor(new Date(dateFrom)
        .setUTCHours(0,0,0,0)/1000) :
      dateFrom,
    typeof dateTo === 'string' ?
      Math.floor(new Date(dateTo)
        .setUTCHours(23,59,59,0)/1000) :
      dateTo
  )

  const hash = computeHash(candles)

  // Save to memory and disk
  GLOBAL_CANDLE_CACHE.set(key, {
    candles, fetchedAt: Date.now(), hash
  })

  fs.mkdirSync(path.dirname(cacheFile), 
    { recursive: true })
  fs.writeFileSync(cacheFile, JSON.stringify({
    symbol, key, candles,
    fetchedAt: Date.now(), hash
  }))

  console.log(`Fetched and cached: ${key}
    Candles: ${candles.length}
    Hash: ${hash}`)

  return candles
}

function computeHash(candles: Candle[]): string {
  // Always same result for same candles
  const sorted = [...candles].sort(
    (a,b) => a.time - b.time
  )
  const sum = sorted.reduce(
    (s,c) => s + c.close, 0
  )
  return `${sorted.length}:${sum.toFixed(2)}`
}
