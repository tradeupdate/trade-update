async function syncDerivContracts(
  userId: string,
  derivService: DerivService
): Promise<void> {

  console.log(`Syncing contracts for ${userId}`)

  // Step 1: Get open trades from database
  const dbOpenTrades = db.prepare(`
    SELECT * FROM trades 
    WHERE user_id = ? AND status = 'open'
    AND is_paper = 0
  `).all(userId)

  // Step 2: Get open contracts from Deriv
  let derivContracts: any[] = []
  try {
    derivContracts = await derivService
      .getOpenContracts()
    // Deriv API call: { portfolio: 1 }
  } catch (error) {
    console.error('Failed to fetch Deriv 
      portfolio:', error)
    return
  }

  // Step 3: Compare and reconcile
  for (const dbTrade of dbOpenTrades) {
    const derivContract = derivContracts.find(
      c => c.contract_id === dbTrade.contract_id
    )

    if (!derivContract) {
      // Trade in DB but not in Deriv
      // It must have closed during disconnect
      console.warn(`Trade ${dbTrade.id} 
        not found in Deriv — marking closed`)

      db.prepare(`
        UPDATE trades SET 
        status = 'closed',
        closed_at = ?,
        pnl = ?,
        exit_price = ?,
        close_reason = 'closed_during_disconnect'
        WHERE id = ?
      `).run(
        Date.now(),
        0, // Unknown P&L
        dbTrade.entry_price, // Use entry as estimate
        dbTrade.id
      )

      // Send notification
      sendNotification(userId, {
        title: '⚠️ Trade closed during disconnect',
        body: `${dbTrade.direction} trade was 
          closed while bot was disconnected. 
          Check Deriv for final P&L.`
      })
    }
  }

  // Step 4: Check for unknown Deriv contracts
  for (const contract of derivContracts) {
    const dbTrade = dbOpenTrades.find(
      t => t.contract_id === contract.contract_id
    )

    if (!dbTrade) {
      console.warn(`Unknown contract 
        ${contract.contract_id} found in Deriv`)
      
      // Create a sync record
      db.prepare(`
        INSERT INTO trades (
          id, user_id, direction, entry_price,
          stake, status, is_paper, contract_id,
          close_reason, opened_at
        ) VALUES (?, ?, ?, ?, ?, 'open', 0, ?, 
          'synced_from_deriv', ?)
      `).run(
        crypto.randomUUID(),
        userId,
        contract.contract_type === 'CALL' ? 
          'BUY' : 'SELL',
        contract.entry_spot,
        contract.buy_price,
        contract.contract_id,
        Date.now()
      )

      sendNotification(userId, {
        title: '📋 Unknown trade synced',
        body: `Found open contract in Deriv 
          not tracked by bot. Added to history.`
      })
    }
  }

  console.log(`Contract sync complete for 
    ${userId}: ${dbOpenTrades.length} DB trades,
    ${derivContracts.length} Deriv contracts`)
}

app.get('/api/user/dashboard',

ITEM 3 — DERIV API REJECTION HANDLER
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Handle all Deriv API rejections gracefully.

async function placeOrderWithRetry(
  derivService: DerivService,
  direction: 'BUY' | 'SELL',
  stake: number,
  userId: string,
  botEngine: BotEngine
): Promise<OrderResult> {

  const MAX_RETRIES = 1
  const RETRY_DELAY = 3000 // 3 seconds
  const REJECTION_COOLDOWN = 5 * 60 * 1000 
  // 5 minutes extra cooldown after rejection

  for (let attempt = 1; attempt <= MAX_RETRIES + 1; attempt++) {
    try {
      console.log(`Order attempt ${attempt}: 
        ${direction} $${stake}`)

      const result = await derivService.placeOrder(
        direction, stake
      )

      if (result.success) {
        console.log(`Order placed successfully: 
          contract_id=${result.contractId}`)
        return result
      }

      // Order failed
      console.error(`Order rejected on attempt 
        ${attempt}: ${result.error}`)

      if (attempt <= MAX_RETRIES) {
        console.log(`Retrying in 3 seconds...`)
        await new Promise(r => setTimeout(r, RETRY_DELAY))
        continue
      }

      // All attempts failed
      console.error(`All ${MAX_RETRIES + 1} 
        attempts failed. Pausing bot.`)

      // PAUSE THE BOT
      botEngine.pause(
        `Deriv rejected order: ${result.error}`
      )

      // ADD EXTRA COOLDOWN
      botEngine.setExtraCooldown(REJECTION_COOLDOWN)

      // NOTIFY USER
      sendNotification(userId, {
        title: '⚠️ Trade rejected by Deriv',
        body: `${direction} order rejected: 
          ${result.error}. 
          Bot paused — tap to resume.`,
        action: 'open_dashboard'
      })

      // LOG THE REJECTION
      db.prepare(`
        INSERT INTO auth_log (
          id, username, event, details, timestamp
        ) VALUES (?, ?, 'trade_rejected', ?, ?)
      `).run(
        crypto.randomUUID(),
        userId,
        JSON.stringify({ 
          direction, stake, error: result.error,
          attempts: MAX_RETRIES + 1
        }),
        Date.now()
      )

      return { 
        success: false, 
        error: result.error,
        botPaused: true 
      }

    } catch (error) {
      console.error(`Order attempt ${attempt} 
        threw error:`, error)
      
      if (attempt <= MAX_RETRIES) {
        await new Promise(r => setTimeout(r, RETRY_DELAY))
        continue
      }

      botEngine.pause(`Network error placing order`)
      sendNotification(userId, {
        title: '⚠️ Order failed — network error',
        body: 'Bot paused due to network error. 
          Check connection and resume manually.'
      })

      return { success: false, error: String(error) }
    }
  }

  return { success: false, error: 'Max retries exceeded' }
}

ITEM 4 — CAPITAL PRESERVATION MODE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Activates when account balance drops 
below $80 (stake would be below $0.80).

function checkCapitalPreservationMode(
  balance: number,
  currentThreshold: number
): CapitalPreservationResult {

  const PRESERVATION_TRIGGER = 80
  const PRESERVATION_EXIT = 100
  const PRESERVATION_SCORE_THRESHOLD = 23
  // Must score 23+ out of 25

  if (balance <= PRESERVATION_TRIGGER) {
    return {
      active: true,
      stake: 1.00, // Fixed $1 minimum
      scoreThreshold: PRESERVATION_SCORE_THRESHOLD,
      message: `Capital preservation mode — 
        balance $${balance.toFixed(2)} 
        below $${PRESERVATION_TRIGGER}. 
        Fixed $1 stake, score ${PRESERVATION_SCORE_THRESHOLD}+ required.`
    }
  }

  if (balance > PRESERVATION_EXIT) {
    return {
      active: false,
      stake: balance * 0.01,
      scoreThreshold: currentThreshold,
      message: null
    }
  }

  // Between $80 and $100 — still in recovery
  // but use the actual calculated stake
  return {
    active: balance < PRESERVATION_TRIGGER,
    stake: Math.max(1.00, balance * 0.01),
    scoreThreshold: currentThreshold,
    message: null
  }
}

When balance hits $50 (50% drawdown from $100):

function checkHardStop(
  balance: number,
  peakBalance: number,
  botEngine: BotEngine,
  userId: string
): void {

  const HARD_STOP_AMOUNT = 50
  const HARD_STOP_DRAWDOWN = 0.50
  const AUTO_RESTART_HOURS = 24

  const drawdownFromPeak = 
    (peakBalance - balance) / peakBalance
  const belowHardStop = balance <= HARD_STOP_AMOUNT

  if (belowHardStop || drawdownFromPeak >= 0.50) {
    console.log(`HARD STOP TRIGGERED:
      Balance: $${balance}
      Peak: $${peakBalance}
      Drawdown: ${(drawdownFromPeak*100).toFixed(1)}%`)

    // Stop the bot completely
    botEngine.hardStop()

    // Schedule auto-restart after 24 hours
    const restartTime = Date.now() + 
      (AUTO_RESTART_HOURS * 60 * 60 * 1000)

    db.prepare(`
      UPDATE users SET 
      auto_restart_at = ?,
      bot_hard_stopped = 1
      WHERE id = ?
    `).run(restartTime, userId)

    // Send urgent notification
    sendNotification(userId, {
      title: '🛑 HARD STOP — Bot stopped',
      body: `Balance reached $${balance.toFixed(2)}.
        Maximum drawdown hit. Bot stopped.
        Auto-restart in 24 hours at $1 fixed stake.
        Check your account immediately.`,
      priority: 'high'
    })

    // Show full screen alert on dashboard
    broadcastSSE(userId, {
      type: 'hard_stop',
      payload: {
        balance,
        peakBalance,
        drawdown: drawdownFromPeak,
        restartAt: restartTime,
        message: `Bot stopped at $${balance}. 
          Will auto-restart in 24 hours with 
          micro-stake recovery mode.`
      }
    })
  }
}

Recalculate stake daily at midnight UTC.

function runDailyCompound(userId: string): void {
  const user = db.prepare(
    'SELECT * FROM users WHERE id = ?'
  ).get(userId)

  const balance = user.account_balance
  const phase = getGrowthPhase(balance)

  let riskPercent: number
  switch(phase) {
    case 'SEED':      // $0-$200
      riskPercent = 0.01; break
    case 'BUILDING':  // $200-$500
      riskPercent = 0.015; break
    case 'GROWTH':    // $500-$1000
      riskPercent = 0.02; break
    case 'PROTECTION': // $1000+
      riskPercent = 0.015; break
    default:
      riskPercent = 0.01
  }

  const newStake = Math.max(1.00, 
    Math.min(50.00, balance * riskPercent)
  )

  const previousStake = user.current_stake || 1.00

  db.prepare(`
    UPDATE users SET 
    current_stake = ?,
    daily_start_balance = ?,
    risk_percent = ?
    WHERE id = ?
  `).run(newStake, balance, riskPercent, userId)

  if (newStake !== previousStake) {
    sendNotification(userId, {
      title: '💹 Stake updated',
      body: `Daily compound: stake changed from 
        $${previousStake.toFixed(2)} to 
        $${newStake.toFixed(2)} 
        (${(riskPercent*100).toFixed(1)}% of 
        $${balance.toFixed(2)})`
    })
  }

  console.log(`Daily compound for ${user.username}:
    Balance: $${balance}
    Phase: ${phase}
    Risk: ${(riskPercent*100).toFixed(1)}%
    New stake: $${newStake}`)
}

// Schedule at midnight UTC every day:
function scheduleDailyCompound(): void {
  function msUntilMidnightUTC(): number {
    const now = new Date()
    const midnight = new Date()
    midnight.setUTCHours(24, 0, 0, 0)
    return midnight.getTime() - now.getTime()
  }

  setTimeout(() => {
    // Run compound for all active users
    const activeUsers = db.prepare(`
      SELECT id FROM users 
      WHERE is_active = 1 
      AND auto_compound_enabled = 1
    `).all()

    activeUsers.forEach(u => 
      runDailyCompound(u.id)
    )

    // Schedule next midnight
    scheduleDailyCompound()
  }, msUntilMidnightUTC())

  console.log(`Daily compound scheduled for 
    midnight UTC in 
    ${(msUntilMidnightUTC()/3600000).toFixed(1)} hours`)
}

ITEM 7 — CONVICTION-BASED STAKE SCALING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

On highest-conviction setups size up within
safe limits.

function calculateConvictionStake(
  baseStake: number,
  score: number,
  maxScore: number,
  balance: number
): number {

  const scorePercent = score / maxScore

  let multiplier: number
  if (scorePercent >= 0.96) {
    // 96%+ of max score = maximum conviction
    multiplier = 1.5
  } else if (scorePercent >= 0.88) {
    multiplier = 1.25
  } else if (scorePercent >= 0.80) {
    multiplier = 1.0
  } else if (scorePercent >= 0.72) {
    multiplier = 0.75
    // Below-threshold quality = smaller stake
  } else {
    multiplier = 1.0 // Default
  }

  const convictionStake = baseStake * multiplier

  // Hard caps always enforced
  const maxAllowed = Math.min(
    50.00,           // Absolute maximum
    balance * 0.02   // Maximum 2% of balance
  )

  const finalStake = Math.max(
    1.00,
    Math.min(convictionStake, maxAllowed)
  )

  if (multiplier !== 1.0) {
    console.log(`Conviction sizing:
      Score: ${score}/${maxScore} 
        (${(scorePercent*100).toFixed(0)}%)
      Base stake: $${baseStake}
      Multiplier: ${multiplier}×
      Final stake: $${finalStake}`)
  }

  return finalStake
}
