function isInActiveSession(unixTimestampSeconds: number): boolean {
  // Convert to UTC hour
  const date = new Date(unixTimestampSeconds * 1000)
  const utcHour = date.getUTCHours()
  const utcMinutes = date.getUTCMinutes()
  const utcDecimalHour = utcHour + (utcMinutes / 60)

  // Asian session: 00:00 - 03:00 UTC
  const asian = utcDecimalHour >= 0 && utcDecimalHour < 3

  // London Open: 07:00 - 10:00 UTC
  const london = utcDecimalHour >= 7 && utcDecimalHour < 10

  // London/NY Overlap: 12:00 - 16:00 UTC
  const overlap = utcDecimalHour >= 12 && utcDecimalHour < 16

  // NY Afternoon: 13:00 - 17:00 UTC
  const ny = utcDecimalHour >= 13 && utcDecimalHour < 17

  const result = asian || london || overlap || ny

  return result
}

// EMA validation
function safeEMA(candles: Candle[], period: number): number {
  if (!candles || candles.length < period) {
    return candles?.[candles.length - 1]?.close || 0
  }
  const closes = candles.map(c => c.close)
  if (closes.some(v => isNaN(v) || v <= 0)) {
    console.warn(`Invalid closes in EMA calculation`)
    return closes[closes.length - 1] || 0
  }
  const k = 2 / (period + 1)
  let ema = closes.slice(0, period).reduce((a, b) => a + b, 0) / period
  for (let i = period; i < closes.length; i++) {
    ema = closes[i] * k + ema * (1 - k)
  }
  if (isNaN(ema)) {
    console.warn(`EMA${period} resulted in NaN`)
    return closes[closes.length - 1]
  }
  return ema
}

// RSI validation
function safeRSI(candles: Candle[], period: number = 14): number {
  if (!candles || candles.length < period + 1) return 50
  
  const closes = candles.map(c => c.close)
  let gains = 0, losses = 0
  
  for (let i = 1; i <= period; i++) {
    const diff = closes[i] - closes[i - 1]
    if (diff > 0) gains += diff
    else losses += Math.abs(diff)
  }
  
  if (gains === 0 && losses === 0) return 50
  
  let avgGain = gains / period
  let avgLoss = losses / period
  
  for (let i = period + 1; i < closes.length; i++) {
    const diff = closes[i] - closes[i - 1]
    avgGain = (avgGain * (period - 1) + Math.max(0, diff)) / period
    avgLoss = (avgLoss * (period - 1) + Math.max(0, -diff)) / period
  }
  
  if (avgLoss === 0) return 100
  const rs = avgGain / avgLoss
  const rsi = 100 - (100 / (1 + rs))
  
  if (isNaN(rsi)) return 50
  return Math.max(0, Math.min(100, rsi))
}

// ATR validation
function safeATR(candles: Candle[], period: number = 14): number {
  if (!candles || candles.length < 2) return 100 // default 100 pips
  
  const trs: number[] = []
  for (let i = 1; i < candles.length; i++) {
    const tr = Math.max(
      candles[i].high - candles[i].low,
      Math.abs(candles[i].high - candles[i-1].close),
      Math.abs(candles[i].low - candles[i-1].close)
    )
    if (!isNaN(tr) && tr >= 0) trs.push(tr)
  }
  
  if (trs.length === 0) return 100
  
  const usePeriod = Math.min(period, trs.length)
  let atr = trs.slice(0, usePeriod).reduce((a,b) => a+b, 0) / usePeriod
  
  const k = 2 / (usePeriod + 1)
  for (let i = usePeriod; i < trs.length; i++) {
    atr = trs[i] * k + atr * (1 - k)
  }
  
  if (isNaN(atr) || atr <= 0) return 100
  return atr
}

// Bollinger Bands validation
function safeBB(candles: Candle[], period: number = 20, mult: number = 2) {
  if (!candles || candles.length < period) {
    const price = candles?.[candles.length-1]?.close || 39000
    return { 
      upper: price * 1.002, 
      middle: price, 
      lower: price * 0.998,
      width: 0.4
    }
  }
  
  const closes = candles.slice(-period).map(c => c.close)
  const sma = closes.reduce((a,b) => a+b, 0) / period
  const variance = closes.reduce((sum, c) => 
    sum + Math.pow(c - sma, 2), 0) / period
  const std = Math.sqrt(variance)
  
  if (isNaN(sma) || isNaN(std)) {
    const price = candles[candles.length-1].close
    return { upper: price*1.002, middle: price, lower: price*0.998, width: 0.4 }
  }
  
  return {
    upper: sma + (mult * std),
    middle: sma,
    lower: sma - (mult * std),
    width: ((mult * 2 * std) / sma) * 100
  }
}

function determineSignal(
  trendDirection: string,
  marketMode: string,
  rsi: number,
  scores: ComponentScores,
  totalScore: number,
  threshold: number
): "BUY" | "SELL" | "NONE" {

  // Must meet minimum score threshold
  if (totalScore < threshold) return "NONE"

  if (marketMode === "TRENDING" || marketMode === "MIXED") {
    // In trending mode, follow the trend
    if (trendDirection === "BULL" && rsi < 65) {
      return "BUY"
    }
    if (trendDirection === "BEAR" && rsi > 35) {
      return "SELL"
    }
  }

  if (marketMode === "OSCILLATING") {
    // In oscillating mode, trade extremes
    if (rsi < 30) return "BUY"
    if (rsi > 70) return "SELL"
  }

  // Counter-trend at RSI extremes
  if (rsi < 22) return "BUY"
  if (rsi > 78) return "SELL"

  return "NONE"
}
